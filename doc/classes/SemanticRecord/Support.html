attr_accessor_with_versioning(*attributes)

replacement for the standard attr_accessor. extends it to behave like an versioning system, to keep track of old values after changes

[Source]

    # File lib/semantic_record/support.rb, line 6
 6:       def attr_accessor_with_versioning(*attributes)
 7:          # inspired from: http://www.archivum.info/comp.lang.ruby/2008-04/msg03767.html
 8:          attributes.each { |attribute|
 9:                    module_eval %{
10:                            def #{attribute} (version = :actual)
11:                              if version == :actual
12:                                @#{attribute}
13:                              elsif version == :old
14:                                @#{attribute}_old
15:                              else
16:                                raise ArgumentError, "unkown access symbol"
17:                              end
18:                            end
19:                            
20:                            def set_#{attribute} (value, init = false)
21:                              if init
22:                                @#{attribute}_modified = nil
23:                                @#{attribute}_old = nil
24:                              end
25:                              if @#{attribute}_modified == nil
26:                                 @#{attribute}_modified = 0
27:                              elsif @#{attribute}_modified == 1
28:                                @#{attribute}_old = @#{attribute}
29:                              end
30:                              @#{attribute} = value
31:                              @#{attribute}_modified += 1
32:                            end
33:                            
34:                            def #{attribute}= (value)
35:                              set_#{attribute} value
36:                            end
37:                    }
38:            }
39:       end
